import numpy as np
import matplotlib.pyplot as plt

class RK4Solver:
    def __init__(self, func, dt):
        self.func = func  # 导数函数 f(t, y)
        self.dt = dt      # 时间步长

    def step(self, t, y):
        dt = self.dt
        f = self.func

        k1 = f(t, y)
        k2 = f(t + 0.5 * dt, y + 0.5 * dt * k1)
        k3 = f(t + 0.5 * dt, y + 0.5 * dt * k2)
        k4 = f(t + dt, y + dt * k3)

        return y + (dt / 6.0) * (k1 + 2*k2 + 2*k3 + k4)

# SHO 的导数函数 f(t, y)
def sho_derivative(t, y, omega=1.0):
    x, v = y          # y = [x, v]
    dxdt = v          # dx/dt = v
    dvdt=-omega**2 * x  # dv/dt = -ω²x
    return np.array([dxdt, dvdt])

# 参数设置
omega = 2.0          # 角频率
y0 = np.array([1.0, 0.0])  # 初始条件 [x=1, v=0]
t_start, t_end = 0, 10
dt = 0.01
t_values = np.arange(t_start, t_end, dt)

# 初始化求解器
solver = RK4Solver(lambda t, y: sho_derivative(t, y, omega), dt)

# 数值求解
y_values = np.zeros((len(t_values), 2))
y_values[0] = y0
for i in range(1, len(t_values)):
    y_values[i] = solver.step(t_values[i-1], y_values[i-1])

# 解析解（对比）
x_exact = y0[0] * np.cos(omega * t_values)

# 绘图
plt.plot(t_values, y_values[:, 0], label="RK4 (Numerical)")
plt.plot(t_values, x_exact, "--", label="Exact Solution")
plt.xlabel("Time")
plt.ylabel("Position (x)")
plt.legend()
plt.grid()
plt.show()
